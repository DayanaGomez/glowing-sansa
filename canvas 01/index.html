Drawing a Rectangle

html

<canvas id= "mycanvas" width= "600px" height= "400px" ></canvas>

css

mycanvas { background: white; border: 1px solid black;}

js

var c = document.getElementById("mycanvas") ;

var ctx = c.getContext("2d");

ctx.fillStyle = "#00FFFF";

ctx.fillRect (10,20,150,75);

.

Drawing a Line

html

<canvas id= "mycanvas" width= "600px" height= "400px" ></canvas>

css

mycanvas { background: white; border: 1px solid black;}

js

var c = document.getElementById("mycanvas") ;

var ctx = c.getContext("2d");

ctx.moveTo (0,0);

ctx.lineTo (200,400);

ctx.stroke();

Drawing a Circle

html

<canvas id= "mycanvas" width= "600px" height= "400px" ></canvas>

css

mycanvas { background: white; border: 1px solid black;}

js

var c = document.getElementById("mycanvas") ;

var ctx = c.getContext("2d");

ctx.beginPath ();

ctx.arcTo (95,50,40,0.2);

ctx.stroke();

.

Drawing a Gradient

js

var c = document.getElementById("mycanvas") ;

var ctx = c.getContext("2d");

var grd = ctx.createLinearGradient(0,0,200,0);

grd.addColorStop (0,"black");

grd.addColorStop (1,"white");

ctx.fillStyle = grd;

ctx.fillRect (10,10,150,80);

.

Adding Color and Opacity

js

var c = document.getElementById("mycanvas") ;

var ctx = c.getContext("2d");

ctx.fillStyle = "#blue";

ctx.fillRect (10,20,150,75);

ctx.beginPath ();

ctx.arcTo (95,50,40,0.2);

ctx.globalAlpha();

ctx.stroke();
